server {
    listen [::]:443 ssl;
    listen 443 ssl;
    server_name www.sushizlyubovyu.com.ua;

    ssl_certificate /etc/letsencrypt/live/sushizlyubovyu.com.ua/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/sushizlyubovyu.com.ua/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        try_files $uri $uri/ /index.html;
    }

    location /backend/ {
        proxy_pass http://localhost:1234/;
        # додайте інші налаштування проксі, які вам потрібні
    }
}
